def ad2geo(address, api_key):
  params = dict(service="address", request="getcoord", version="2.0", crs="epsg:4326", refine="true", simple="false", key=api_key)
  params['type'] = "ROAD"
  params['address'] = address
  resp = req.get(api_url, params=params).json()
  print(resp)
  if resp['response']['status'] == 'OK':
    return resp['response']['result']['point']['x'], resp['response']['result']['point']['y'] 
  else:
    return 0,0
